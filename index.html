bcralnit.js

jQuery plugin to add line number in textarea.

Sample:

// Default
$("textarea").bcralnit();

// with options
$("textarea").bcralnit({
    width: '50px',           // default: '60px'
    background: 'salmon',    // default: '#ddd'
    color: '#fff'            // default: '#333'
});
